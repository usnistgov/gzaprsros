171 public:
172  IkSolution(const std::vector<IkSingleDOFSolutionBase<T> > & vinfos, const std::vector<int> & vfree)

178  virtual void GetSolution (T *solution, const T *freevalues) const
179  {
180  for ( std::size_t i = 0; i < _vbasesol.size( ); ++i )
181  {
182  if ( _vbasesol[i].freeind < 0 )
183  {
184  solution[i] = _vbasesol[i].foffset;
185  }
186  else
187  {
188  solution[i] = freevalues[_vbasesol[i].freeind] * _vbasesol[i].fmul + _vbasesol[i].foffset;
189 
190  if ( solution[i] > T(3.14159265358979) )
191  {
192  solution[i] -= T(6.28318530717959);
193  }
194  else if ( solution[i] < T(-3.14159265358979) )
195  {
196  solution[i] += T(6.28318530717959);
197  }
198  }
199  }
200  }

218  virtual void Validate ( ) const
219  {
220  for ( size_t i = 0; i < _vbasesol.size( ); ++i )
221  {
222  if ( _vbasesol[i].maxsolutions == (unsigned char) -1 )
223  {
224  throw std::runtime_error("max solutions for joint not initialized");
225  }
226 
227  if ( _vbasesol[i].maxsolutions > 0 )
228  {
229  if ( _vbasesol[i].indices[0] >= _vbasesol[i].maxsolutions )
230  {
231  throw std::runtime_error("index >= max solutions for joint");
232  }
233 
234  if ( ( _vbasesol[i].indices[1] != (unsigned char) -1 ) && ( _vbasesol[i].indices[1] >= _vbasesol[i].maxsolutions ) )
235  {
236  throw std::runtime_error("2nd index >= max solutions for joint");
237  }
238  }
239 
240  if ( !std::isfinite(_vbasesol[i].foffset) )
241  {
242  throw std::runtime_error("foffset was not finite");
243  }
244  }
245  }

247  virtual void GetSolutionIndices (std::vector<unsigned int> & v) const
248  {
249  v.resize(0);
250  v.push_back(0);
251 
252  for ( int i = (int) _vbasesol.size( ) - 1; i >= 0; --i )
253  {
254  if ( ( _vbasesol[i].maxsolutions != (unsigned char) -1 ) && ( _vbasesol[i].maxsolutions > 1 ) )
255  {
256  for ( size_t j = 0; j < v.size( ); ++j )
257  {
258  v[j] *= _vbasesol[i].maxsolutions;
259  }
260  size_t orgsize = v.size( );

263  {

266  v.push_back(v[j] + _vbasesol[i].indices[1]);
267  }
268  }
269 
270  if ( _vbasesol[i].indices[0] != (unsigned char) -1 )
271  {
272  for ( size_t j = 0; j < orgsize; ++j )
273  {
